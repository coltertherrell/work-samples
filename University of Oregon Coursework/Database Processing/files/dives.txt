<script>
	$(document).ready(function(){
		$("#site_search").hide();
		$("#wild_search").hide();
		
		$("#site").click(function(){
			$("#wild_search").hide();
			$("#site_search").show();
		})
		
		$("#wild").click(function(){
			$("#wild_search").show();
			$("#site_search").hide();
		})
		
		$("#searchSite").click(function(){
			
			var distance = $("#dist").val();
			var depth = $("#depth").val();
			var temp = $("#temp").val();
			
			
			if (distance.length == 0)
				distance = 100000;
			else
				distance = parseInt(distance);
			if (temp.length == 0)
				temp = -100000;
			else
				temp = parseInt(temp);
			if (depth.length == 0)
				depth = 100000;
			else
				depth = parseInt(depth);
				
			$("#results").html("");
			
			$.ajax({
			url : "php/getSites.php",
			type : "GET",
			data : {"dist" : distance, "temp" : temp, "depth" : depth},
			success: function(result){
				$("#results").append(result);
			},
			error: function(result){
				$("#results").append(result)
			}
		});
			
		
		})
	});
</script>
<script>
	function get_sites(wild){
		$("#results").html("");
		$.ajax({
			url : "php/getSitesByWild.php",
			type : "GET",
			data : {"wild" : wild},
			success: function(result){
				$("#results").append(result);
			},
			error: function(result){
				$("#results").append(result)
			}
		});
	}

</script>
