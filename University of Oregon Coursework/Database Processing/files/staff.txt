<script>
	function get_instr(cls){
		$("#results").html("");
		$.ajax({
			url : "php/getStaff.php",
			type : "GET",
			data : {"staff" : cls},
			success: function(result){
				$("#results").append(result);
			},
			error: function(result){
				$("#results").append(result)
			}
		});
	}
