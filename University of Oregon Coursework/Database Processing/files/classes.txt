<script>
	function get_classes(cls){
		$("#results").html("");
		$.ajax({
			url : "php/getClasses.php",
			type : "GET",
			data : {"class" : cls},
			success: function(result){
				$("#results").append("<tr><td><p>CLASS</p></td><td><p>DATE</p></td><td><p>LOCATION</p></td><td><p>COST</p></td></tr>")
				$("#results").append(result);
			},
			error: function(result){
				$("#results").append(result)
			}
		});
	}

</script>
